⚠️ Не настроен доступ к базе
'; } else { const sb = supabase.createClient(cfg.url, cfg.key); let currentMarket = null; async function loadMarkets() { const { data } = await sb.from('markets').select('*').eq('active', true); const el = document.getElementById('markets'); el.innerHTML = (data || []).map(m => `` ).join(''); el.querySelectorAll('.market-btn').forEach(btn => { btn.addEventListener('click', () => selectMarket(btn.dataset.market)); }); if (data && data.length > 0) selectMarket(data[0].id); } async function loadServices() { if (!currentMarket) return; const { data } = await sb .from('services') .select('*') .eq('market_id', currentMarket) .eq('available', true); const el = document.getElementById('services'); if (!data || !data.length) { el.innerHTML = '
Пока нет услуг в этом направлении
'; return; } el.innerHTML = data.map(s => `
${TYPE_LABEL[s.type] || s.type}
${s.title}
${s.description || ''}
${s.price} ${s.currency}
`).join(''); } function selectMarket(id) { currentMarket = id; document.querySelectorAll('.market-btn').forEach(btn => { btn.classList.toggle('active', btn.dataset.market === id); }); loadServices(); } loadMarkets(); }
